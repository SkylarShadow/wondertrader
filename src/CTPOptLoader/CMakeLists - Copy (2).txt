)

        ${INCS}
)
message(STATUS "LNKS = ${LNKS}")
# 6. 查找 boost_filesystem（在 deps 的 lib 目录里找）
if(MSVC)
    if(CMAKE_BUILD_TYPE MATCHES Debug)
        set(BOOST_FS_LIB ${LNKS}/libboost_filesystem-vc142-mt-gd-x64-1_72.lib)
        
    else() # Release 或其他
        set(BOOST_FS_LIB ${LNKS}/libboost_filesystem-vc142-mt-x64-1_72.lib)

    endif()
endif()

# 调试输出（可选）
message(STATUS "Boost filesystem lib: ${BOOST_FS_LIB}")

if(NOT BOOST_FS_LIB)
    message(FATAL_ERROR "boost_filesystem library not found in ${LNKS}")
endif()
