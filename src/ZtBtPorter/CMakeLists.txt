cmake_minimum_required(VERSION 3.10)

#2. 确定工程名
PROJECT(ZtBtPorter LANGUAGES CXX)
SET(CMAKE_CXX_STANDARD 17)

SET(LIBRARY_OUTPUT_PATH ${CMAKE_BINARY_DIR}/build_${PLATFORM}/${CMAKE_BUILD_TYPE}/bin/ZtBtPorter)

#7. 添加源码
file(GLOB SRCS *.cpp)
file(GLOB HDRS *.h)

IF(MSVC)
	list (APPEND SRCS ../Common/mdump.cpp)
ENDIF()

INCLUDE_DIRECTORIES(${INCS})
LINK_DIRECTORIES(${LNKS})
ADD_LIBRARY(ZtBtPorter SHARED ${SRCS} ${HDRS})

IF(MSVC)
	SET(LIBS
		ZtBtCore
		ZTSTools
		ZTSUtils
	)
ELSE(GNUCC)
	SET(LIBS
		ZtBtCore
		ZTSTools
		ZTSUtils

		dl
		pthread
		boost_filesystem
	)
	if(WIN32)
		LIST(APPEND LIBS iconv)
	ENDIF()
ENDIF()


TARGET_LINK_LIBRARIES(ZtBtPorter ${LIBS})
IF (WIN32)
ELSE (UNIX)
	SET_TARGET_PROPERTIES(ZtBtPorter PROPERTIES
		CXX_VISIBILITY_PRESET hidden
		VISIBILITY_INLINES_HIDDEN 1
        LINK_FLAGS_RELEASE -s)
ENDIF ()
